KeyBoard

KeyBoard RN 数字键盘封装

使用方法#

<KeyBoard
addRef={addRef}
deleteRef={deleteRef}
onValueChange={valueChange}
useDot={false}
showKeyboard={false}
deleteIcon={require("../../images/ic_keybord_delete.png")}
keyboardIcon={require("../../images/ic_keybord_hide.png")}
submitStartColor={"#FF5E40"}
submitEndColor={"#FF3B2D"}
submitText={"出价"}
onHideKeyboard={() => {
console.log("onHideKeyboard");
}}
onSubmitCommit={num => {
console.log("onSubmitCommit", num);
}}
/>

Props#

onValueChange: Function;#

键盘数值变化时回调

const valueChange = (num: number) => {
setKeyBoardNum(num);
};

useDot: boolean;#

是否展示小数点

showKeyboard: boolean;#

是否展示软键盘标签

deleteIcon: ImageSourcePropType;#

删除标签icon

keyboardIcon: ImageSourcePropType;#

键盘标签icon

onSubmitCommit: Function;#

点击确认按钮回调

onSubmitCommit={num => {
console.log("onSubmitCommit", num);
}}

submitStartColor: string | number;#

确认按钮渐变开始颜色

submitEndColor: string | number;#

确认按钮渐变结束颜色

submitText: String;#

确认按钮文案

onHideKeyboard: Function;#

键盘标签点击隐藏回调

onHideKeyboard={() => {
console.log("onHideKeyboard");
}}

addRef: any;#

增加数值Ref回调

const addRef = useRef<any>();

deleteRef: any;#

删除数值Ref回调

const deleteRef = useRef<any>();
