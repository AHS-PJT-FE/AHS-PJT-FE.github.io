BottomSheet

使用方法#

bottomSheetView使用方法如下：

<BottomSheetView
modalVisible={modalVisible}
setVisible={changeVisible}
child={<SheetView modalVisible={modalVisible} setVisible={changeVisible} />}
title={"title"}
closeIcon={require("../../images/ic_sheet_close.png")}
containerBackgroundColor={"white"}
sizeStyle={"max"}
/>

自控件使用方法如下：

function SheetView(props: SheetViewProps) {
return (
<> // 具体内容写在下方，无需在最外层包一层view组件
<View style={{ marginStart: 16, marginEnd: 16, marginTop: 16 }}>
<Text>bottom sheet </Text>
<Text>bottom sheet </Text>
<Text>bottom sheet </Text>
</View>
<View style={{ marginBottom: 10, marginStart: 12, marginEnd: 12 }}>
<UIButton
containerStyle={{
alignItems: "center",
width: "100%"
}}
onPress={() => {
props.setVisible(!props.modalVisible);
}}
>
确定
</UIButton>
</View>
</>
);
}

Props#

modalVisible: boolean;#

控制modal是否展示

setVisible: Function;#

传入切换是否显示的function

child: React.Component;#

自控件

minHeight?: number;#

子控件最小高度

maxHeight?: number;#

子控件最大高度

titleBorder?: number;#

头部border，默认16px

title: string;#

title文案

headerView?: React.Component#

自定义title样式

titleBackground?: ColorValue;#

title背景颜色

closeIcon: ImageSourcePropType;#

关闭按钮图标

containerBackgroundColor?: ColorValue;#

子控件默认背景颜色

containerHeight?: number;#

子控件高度

sizeStyle: "min" | "auto" | "max";#

子控件大小模式，最小，自适应，最大

headerTopView?: React.Component#

底部弹框上面的view，与暗拍出价头部信息类似
