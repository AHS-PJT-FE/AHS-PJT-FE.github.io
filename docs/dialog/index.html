dialog

dialog 即对话框，目前封装了7种使用方式。

示例可以在鲁班项目的example中查看

使用方法#

第一步

// 将Dialog组件挂载到你的组件中
import { Dialog } from "@react-native-pjt/ui-lib";
class example extends Component {
private dialogRef: React.RefObject<any>;
constructor(props){
super(props)
this.dialogRef = React.createRef();
}
render(){
return (
<>
<View>
// 组件代码.......
</View>
// 注意Dialog的位置，会影响到Dialog的显示样式
<Dialog ref={this.dialogRef} />
</>
)
}
}

第二步

// 调用默认的Dialog
onShow = () => {
//显示dialog
const key = DialogRef.current?.show({
cancelable: false,
title: "title",
message:
"message",
buttons: [
{
buttonText: "CANCEL",
onPress: () => {
//隐藏dialog
dialogRef.current?.dismiss(key);
}
},
{
buttonText: "OK",
onPress: () => {}
}
]
}
//调用自定义的dialog
onShow = () => {
//显示dialog
const key = DialogRef.current?.showCustom({
cancelable: false,
view: // 这里写自定义view
<View>......</View>
}
//调用自定义message的dialog
onPress={() => {
const key = DialogRef.current?.showSpecial({
cancelable: true,
customMessage: (
<View>
<View style={{ alignItems: "center", justifyContent: "center", flexDirection: "row" }}>
<Text style={{ color: "#FF69B4" }}>自定义</Text>
<Text style={{ color: "#7B68EE" }}>Message</Text>
</View>
<View style={{ alignItems: "center", justifyContent: "center", flexDirection: "row" }}>
<Text style={{ color:"#FF4500", fontSize:18, textAlign: "center" }}>保留title和button的样式，仅自定义message</Text>
</View>
</View>
),
title: "自定义message",
buttons: [
{
buttonText: "CANCEL",
onPress: () => {}
},
{
buttonText: "OK",
onPress: () => {}
}
]
});
}}

Props#

title: text;#

设置dialog的标题

message: text || element;#

设置dialog的内容

buttons: [{button: text, onPress: text}];#

设置dialog按钮的数量，文字，以及点击事件

cancelable: Boolean;#

点击dialog背景是否关闭dialog

view?: element || string;#

该属性仅在showCustom中使用，设置dialog的内容(该模式不保留title和button，需自己编写样式)

customMessage?: element || string;#

该属性仅在showSpecial中使用，设置dialog的内容(该模式保留了title和button，仅需编写message的内容)
