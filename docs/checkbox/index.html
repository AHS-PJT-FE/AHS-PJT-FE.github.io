Checkbox

checkbox 单选组件，状态分为已选中、未选中和禁用状态

使用方法#

const CheckBoxDemoScreen = () => {
return (
<View>
<CheckBox
width={30}
height={30}
selectedIcon={require("../../images/checkbox_selected.png")}
unSelectedIcon={require("../../images/checkbox_unselected.png")}
disableIcon={require("../../images/checkbox_disable.png")}
disable={false}
selected={false}
onChange={checked => {
console.log("checked --------> " + checked);
}}
/>
</View>
);
};

Props#

width: number;#

控件长度

height: number;#

控件高度

selectedIcon: ImageSourcePropType;#

选中图片

unSelectedIcon: ImageSourcePropType;#

未选中图片

disableIcon: ImageSourcePropType;#

禁用图片

disable: Boolean;#

是否禁用状态

selected: Boolean;#

是否选中状态

onChange: (checked: Boolean) => {};#

点击回调事件
