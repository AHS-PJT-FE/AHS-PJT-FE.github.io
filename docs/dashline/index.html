DashLine

使用方法#

<View style={{ flexDirection: "row", width: "100%", height: 50, justifyContent: "center" }}>
<DashLine
dashLineWidth={4}
dashLineHeight={1}
dashLineBorder={0}
dashLineOrientation={"horizontal"}
dashLineColor={"red"}
dashLineSpace={4}
dashLineContainerStyle={{ marginVertical: 0 }}/>
</View>
<View style={{ width: 100, height: 50, justifyContent: "center", alignSelf: "center" }}>
<DashLine
dashLineWidth={4}
dashLineHeight={1}
dashLineBorder={0}
dashLineOrientation={"vertical"}
dashLineColor={"red"}
dashLineSpace={4}
dashLineContainerStyle={{ marginHorizontal: 10 }}/>
</View>

Props#

dashLineWidth: number;#

虚线长度

dashLineHeight: number;#

虚线高度

dashLineBorder: number;#

虚线圆角

dashLineOrientation: "vertical" | "horizontal";#

虚线排列方向 vertical：横向排列 horizontal：纵向排列

dashLineColor: ColorValue;#

虚线颜色

dashLineSpace: number;#

虚线之间间隔距离

dashLineContainerStyle?: ViewStyle;#

虚线外部容器样式
