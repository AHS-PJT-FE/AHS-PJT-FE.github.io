InspectionReport

InspectionReport 质检报告组件，可通过算法动态排序已达到最少空间内展示内容最多；

使用方法#

<InspectionReport
dataList = {data}
warnIcon = {require("../../images/ic_inspection_warn.png"
)
}
picIcon = {require("../../images/ic_inspection_pic.png"
)
}
normalIcon = {require("../../images/toBuySuccessIcon.png"
)
}
onPicClick = {(inspection
:
InspectionReportItemData
) =>
{
console.log("picList", inspection.id);
}
}
/>

具体参数#

// props参数
interface InspectionReportProp {
dataList: Array<InspectionReportData>; // 质检报告列表数据
warnIcon: ImageSourcePropType; // 异常项按钮
normalIcon: ImageSourcePropType; // 正常项按钮
picIcon: ImageSourcePropType; // 质检报告图片按钮
onPicClick?: (inspectionReportItemData?: InspectionReportItemData) => void; // 图片点击回调
marginHorizontal: number; // 质检报告左右边距
}
// 质检报告列表数据结构
export type InspectionReportData = {
dataItem: Array<InspectionReportItemData>; // ppv列表数据
title: string; // ppn 名称
};
// ppv列表数据
export type InspectionReportItemData = {
id: string; // ppv ID
name: string; // ppv名称
picList?: Array<string>; // 图片列表
type: number; // ppv类型，是否异常项; 1 - 异常项 0 - 正常项
sort: number; // 排序字段，是否优先展示; 1 - 优先展示 0 - 正常排序
};
